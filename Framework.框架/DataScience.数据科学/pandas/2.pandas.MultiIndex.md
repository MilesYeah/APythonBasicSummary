# pandas.MultiIndex

```py
>>> s1 = Series(np.random.randn(6), index=[['1','1','1','2','2','2'], ['a','b','c','a','b','c']])
>>> s1
1  a    0.759539
   b    1.670224
   c    0.718121
2  a    1.436442
   b   -1.991585
   c    0.425283
dtype: float64
>>> s1['1']
a    0.759539
b    1.670224
c    0.718121
dtype: float64
>>> s1['2']
a    1.436442
b   -1.991585
c    0.425283
dtype: float64
```

```py
>>> s1['1']['a']
0.7595389264707274
>>> s1[:, 'a']
1    0.759539
2    1.436442
dtype: float64
>>> type(s1['1'])
<class 'pandas.core.series.Series'>
```
```py
>>> df = s1.unstack()
>>> df
          a         b         c
1  0.759539  1.670224  0.718121
2  1.436442 -1.991585  0.425283

>>> df1 = DataFrame([s1['1'], s1['2']])
>>> df1
          a         b         c
0  0.759539  1.670224  0.718121
1  1.436442 -1.991585  0.425283

```
