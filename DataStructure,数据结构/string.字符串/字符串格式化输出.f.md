# 字符串 输出格式控制


## f string

```py
f = 1.0123456789

print(f"{f}")
1.0123456789

print(f"{f:>20}")
        1.0123456789

print(f"{f:<20}")
1.0123456789        

print(f"{f:^20}")
    1.0123456789    

print(f"{f:>20.6}")
             1.01235

print(f"{f:>20.8}")
           1.0123457

print(f"{f:<20.6}")
1.01235             

print(f"{f:<20.8}")
1.0123457           

print(f"{f:^20.8}")
     1.0123457      

print(f"{f:0>20.8}")
000000000001.0123457

print(f"{f:*>20.8}")
***********1.0123457

print(f"{f:*<20.8}")
1.0123457***********

print(f"{f:*^20.8}")
*****1.0123457******
```
